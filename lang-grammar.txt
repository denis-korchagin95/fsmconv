special characters:
	@empty - it represents an 'epsilon' transition to state

directives:
	start - it allows mark states which represent an initial states
	end - it allows mark states which represent a final states

unit: statement_list;

statement_list: statement
	| statement, statement_list
	;

statement: directive | nfa-rule ;

nfa-rule: nfa-transition by character_list ;

nfa-transition: nfa-state to nfa-state
	| nfa-state -> nfa-state
	;

nfa-state: identifier ;

directive: keyword identifier_list;

keyword: start
	| end
	| to
	| by
	;

identifier_list: identifier
	| identifier, identifier_list
	;

character_list: character
	| special-character
	| character, character_list
	| special-character, character_list
	;

special-character: @ identifier;

identifier: identifier-start-letter;

identifier-start-letter: [a-zA-Z_]
	| [a-zA-Z_] identifier-letter
	;

identifier-letter: [a-zA-Z0-9_]
	| [a-zA-Z0-9_] identifier-letter
	;

character: ' character-letter ';


character-letter:
		any member of the source character set except the single-quote, backslash, or new-line charater
	|	escape-sequence
	;

escape-sequence: \', \\;
